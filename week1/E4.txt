class Solution {
    public int maxProfit(int[] prices) {
        int n = prices.length, profit = 0, minPrice = prices[0];
        int i;
        for(i = 1; i < n; i++){
            minPrice = Math.min(minPrice, prices[i]);
            profit = Math.max(profit, prices[i] - minPrice);
        }
      return profit;
    }
}
